
#include <iostream>
using namespace std;

int main() {
    char exp[100], stack[100];
    int top = -1;

    cout << "Enter an expression: ";
    cin >> exp;

    for (int i = 0; exp[i] != '\0'; i++) {
        char ch = exp[i];

   
        if (ch == '(' || ch == '{' || ch == '[') {
            stack[++top] = ch;
        }
       
        else if (ch == ')' || ch == '}' || ch == ']') {
            if (top == -1) {  
                cout << "Not Balanced\n";
                return 0;
            }
            char last = stack[top--];
            if ((ch == ')' && last != '(') ||
                (ch == '}' && last != '{') ||
                (ch == ']' && last != '[')) {
                cout << "Not Balanced\n";
                return 0;
            }
        }
    }

    // If stack empty â†’ balanced
    if (top == -1)
        cout << "Balanced\n";
    else
        cout << "Not Balanced\n";

    return 0;
}
