#include <iostream>
using namespace std;

#define MAX 100

int main() {
    int n, arr[MAX], stack[MAX], ans[MAX];
    int top = -1;

    cout << "Enter size of array: ";
    cin >> n;
    cout << "Enter elements: ";
    for (int i = 0; i < n; i++) cin >> arr[i];

    
    for (int i = n - 1; i >= 0; i--) {
        while (top != -1 && stack[top] <= arr[i]) {
            top--;  
        }
        if (top == -1) ans[i] = -1;
        else ans[i] = stack[top];
        stack[++top] = arr[i]; 
    }

    cout << "Next Greater Elements: ";
    for (int i = 0; i < n; i++) cout << ans[i] << " ";
    cout << endl;

    return 0;
}
