#include <iostream>
#include <stack>
using namespace std;

int main() {
    int n;
    cin >> n;
    int A[100], B[100];

    for (int i = 0; i < n; i++)
        cin >> A[i];

    for (int i = 0; i < n; i++)
        B[i] = A[i];

    for (int i = 0; i < n - 1; i++)
        for (int j = i + 1; j < n; j++)
            if (B[i] > B[j]) {
                int t = B[i];
                B[i] = B[j];
                B[j] = t;
            }

    stack<int> s;
    int j = 0;
    for (int i = 0; i < n; i++) {
        s.push(A[i]);
        while (!s.empty() && s.top() == B[j]) {
            s.pop();
            j++;
        }
    }

    if (j == n)
        cout << "Yes";
    else
        cout << "No";
}
